# TimeXer 模型完整训练命令 (Windows PowerShell)

# ============================================
# 完整参数训练命令
# ============================================
python run.py `
  --task_name long_term_forecast `                     # 任务名称：长期预测
  --is_training 1 `                                    # 是否训练模式：1=训练，0=测试
  --root_path "dataset/bullet/category/" `            # 数据根目录路径
  --data_path ".300BLK.csv" `                          # 数据文件名
  --model_id "300BLK_merged" `                         # 模型标识符（用于区分不同实验）
  --model TimeXer `                                    # 模型名称
  --data custom `                                      # 数据集类型
  --features M `                                       # 预测任务类型：M=多变量预测多变量，S=单变量预测单变量，MS=多变量预测单变量
  --target ".300BLK 5" `                               # 目标变量（当features为S或MS时使用）
  --freq "h" `                                         # 时间特征编码频率：s=秒级，t=分钟级，h=小时级，d=天级
  --checkpoints "./checkpoints/" `                     # 模型检查点保存路径
  --seq_len 96 `                                       # 输入序列长度
  --label_len 48 `                                     # 起始标记长度（用于解码器）
  --pred_len 96 `                                      # 预测序列长度
  --seasonal_patterns "Monthly" `                      # 季节性模式（M4数据集使用）
  --inverse False `                                    # 是否反转输出数据
  --expand 2 `                                         # Mamba模型扩展因子
  --d_conv 4 `                                         # Mamba模型卷积核大小
  --top_k 5 `                                          # TimesBlock中的top-k参数
  --num_kernels 6 `                                    # Inception中的卷积核数量
  --enc_in 3 `                                         # 编码器输入维度（特征数）
  --dec_in 3 `                                         # 解码器输入维度
  --c_out 3 `                                          # 输出维度
  --d_model 512 `                                      # 模型隐藏层维度
  --n_heads 8 `                                        # 多头注意力头数
  --e_layers 2 `                                       # 编码器层数
  --d_layers 1 `                                       # 解码器层数
  --d_ff 2048 `                                        # 前馈网络维度
  --moving_avg 25 `                                    # 移动平均窗口大小
  --factor 3 `                                         # 注意力因子
  --distil True `                                      # 是否使用蒸馏
  --dropout 0.1 `                                      # Dropout比率
  --embed "timeF" `                                    # 时间特征编码方式：timeF=时间特征，fixed=固定编码，learned=学习编码
  --activation "gelu" `                                # 激活函数
  --output_attention False `                           # 是否输出编码器的注意力权重
  --channel_independence 1 `                           # 通道独立性：0=通道相关，1=通道独立（FreTS模型使用）
  --decomp_method "moving_avg" `                       # 序列分解方法：moving_avg=移动平均，dft_decomp=傅里叶分解
  --use_norm 1 `                                       # 是否使用归一化：1=使用，0=不使用
  --down_sampling_layers 0 `                           # 下采样层数
  --down_sampling_window 1 `                           # 下采样窗口大小
  --down_sampling_method None `                        # 下采样方法：avg=平均池化，max=最大池化，conv=卷积
  --seg_len 48 `                                       # SegRNN的片段长度
  --num_workers 10 `                                   # 数据加载线程数
  --itr 1 `                                            # 实验重复次数
  --train_epochs 10 `                                  # 训练轮数
  --batch_size 32 `                                    # 批大小
  --patience 3 `                                       # 早停耐心值
  --learning_rate 0.0001 `                             # 学习率
  --des "Exp" `                                        # 实验描述
  --loss "MSE" `                                       # 损失函数：MSE=均方误差
  --lradj "type1" `                                    # 学习率调整策略
  --use_amp False `                                    # 是否使用自动混合精度训练
  --use_gpu True `                                     # 是否使用GPU
  --gpu 0 `                                            # GPU设备号
  --use_multi_gpu False `                              # 是否使用多GPU
  --devices "0,1,2,3" `                                # 多GPU设备ID列表
  --p_hidden_dims "[128, 128]" `                       # 去平稳投影器隐藏层维度列表
  --p_hidden_layers 2 `                                # 去平稳投影器隐藏层数
  --use_dtw False `                                    # 是否使用DTW度量（计算耗时，非必要不建议使用）
  --augmentation_ratio 0 `                             # 数据增强比例
  --seed 2 `                                           # 随机种子
  --patch_len 16                                       # TimeXer的patch长度
